# reportData 参数说明文档

本文档描述了天赋进化报告模板中 `reportData` 对象的完整数据结构。

---

## TypeScript 类型定义

```typescript
interface ReportData {
  meta: Meta;
  header: Header;
  hero: Hero;
  diagnosis: Diagnosis;
  talents: Talents;
  synergy: Synergy;
  competitive: Competitive;
  scenario: Scenario;
  action: Action;
  growth: Growth;
  footer: Footer;
  cta: CTA;
}

// ============================================================
// meta - 页面元数据
// ============================================================
interface Meta {
  title: string; // 页面标题，显示在浏览器标签页，如 "闫婷婷 - 天赋进化报告"
}

// ============================================================
// header - 头部用户信息
// ============================================================
interface Header {
  profileLabel: string; // 固定值: "Talent Intelligence Profile"
  name: string; // 用户姓名
  archetype: string; // 天赋类型组合，如 "洞察型 × 韧性型"
  stages: Stages;
}

interface Stages {
  list: Stage[]; // 固定4项: 萌芽期 → 潜能期 → 拥抱期 → 通透期
}

interface Stage {
  label: string; // 阶段名称，固定值依次为: "萌芽期" | "潜能期" | "拥抱期" | "通透期"
}

// ============================================================
// hero - 主图区域
// ============================================================
interface Hero {
  image: HeroImage;
  reportId: string; // 报告唯一标识符，格式如 "YAN-2026-CORE"
}

interface HeroImage {
  url: string; // 主图 URL
  alt: string; // 图片替代文本
  fallbackUrl?: string; // 可选，备用图片 URL（加载失败时使用）
}

// ============================================================
// diagnosis - 阶段诊断
// ============================================================
interface Diagnosis {
  title: string; // 诊断标题，如 "阶段诊断：为何是潜能期？"
  subtitle: string; // 固定值: "Stage Diagnosis Detail"
  basisTitle: string; // 固定值: "诊断依据"
  items: DiagnosisItem[]; // 固定2项
  nextStageAdvice: NextStageAdvice;
}

interface DiagnosisItem {
  index: string; // 序号，如 "01."、"02."
  title: string; // 诊断要点标题（以冒号结尾），如 "他人已见，自己未信："
  desc: string; // 诊断要点描述
}

interface NextStageAdvice {
  title: string; // 固定值: "迈向下一阶段建议"
  lead: string; // 建议标题（加粗显示，以冒号结尾），如 "收集证据："
  desc: string; // 建议详细描述
}

// ============================================================
// talents - 核心天赋识别
// ============================================================
interface Talents {
  title: string; // 固定值: "核心天赋识别"
  subtitle: string; // 固定值: "Core Identification"
  list: TalentItem[]; // 固定3项: 主天赋、辅助天赋、潜在天赋
}

interface TalentItem {
  badge: string; // 徽章文字（单个汉字），如 "洞"、"韧"、"学"
  title: string; // 天赋标题，包含序号，如 "1. 主天赋：洞察型"
  score: TalentScore;
  desc: string; // 天赋描述
  quotes: string[]; // 引用列表（2~3项），用户原话作为佐证
  reason: TitleDesc; // 得分原因
  advice?: TitleDesc; // 可选，验证建议（仅第3项潜在天赋有此字段）
}

interface TalentScore {
  label: string; // 标签: "强度" 或 "待验证"
  value?: string; // 分值，如 "7/10"（待验证时可为空字符串）
}

interface TitleDesc {
  title: string; // 标题（以冒号结尾），如 "为什么是7分："
  desc: string; // 描述内容
}

// ============================================================
// synergy - 天赋组合分析
// ============================================================
interface Synergy {
  title: string; // 固定值: "天赋组合分析"
  subtitle: string; // 固定值: "Synergy Analysis"
  summary: SynergySummary;
  roleSection: RoleSection;
  items: SynergyItem[]; // 固定2项: 推荐角色 + 潜在盲点
}

interface SynergySummary {
  headline: string; // 核心结论（高亮显示），如 "你是「幕后军师」与「苦行僧」的结合体。"
  desc: string; // 详细描述
}

interface RoleSection {
  title: string; // 固定值: "推荐角色定位"
  cards: RoleCard[]; // 固定2项: 核心逻辑 + 环境偏好
  tip: Tip;
}

interface RoleCard {
  label: string; // 卡片标签: "核心逻辑" 或 "环境偏好"
  title: string; // 卡片标题，如 "独立判断 + 独立执行"
  desc: string; // 卡片描述，如 "适合：独立研究者、深度分析师..."
}

interface Tip {
  label: string; // 提示标签，如 "💡 为什么？"
  desc: string; // 提示内容
}

// synergy.items 有两种结构
interface SynergyItemRole {
  badge: string; // 固定值: "职"
  title: string; // 固定值: "推荐角色"
  desc: string; // 推荐角色描述
}

interface SynergyItemBlindSpot {
  badge: string; // 固定值: "盲"
  title: string; // 固定值: "潜在盲点"
  desc: string[]; // 盲点描述列表（固定2项）
}

type SynergyItem = SynergyItemRole | SynergyItemBlindSpot;

// ============================================================
// competitive - 比较优势定位
// ============================================================
interface Competitive {
  title: string; // 固定值: "比较优势定位"
  subtitle: string; // 固定值: "Competitive Edge"
  table: CompetitiveTable;
  card: CompetitiveCard;
}

interface CompetitiveTable {
  columns: string[]; // 固定值: ["能力维度", "类型", "判断依据"]
  rows: CompetitiveTableRow[]; // 固定4项
}

interface CompetitiveTableRow {
  label: string; // 能力维度名称，如 "想得明白"
  type: string; // 类型: "天赋" | "优势" | "避坑"
  reason: string; // 判断依据
}

interface CompetitiveCard {
  label: string; // 固定值: "Who I Am Better Than"
  prefix: string; // 前缀文本，如 "你不需要成为最老练的专家，你只需成为那个"
  highlight: string; // 高亮文本，如 ""比对方想得更清楚、更透彻的人""
  suffix: string; // 后缀文本，如 "。"
}

// ============================================================
// scenario - 场景适配矩阵
// ============================================================
interface Scenario {
  title: string; // 固定值: "场景适配矩阵"
  subtitle: string; // 固定值: "Scenario Matrix"
  items: ScenarioItem[]; // 固定3项: A级(推荐) → B级(保持) → D级(避免)
  details: ScenarioDetail[]; // 固定2项
}

interface ScenarioItem {
  badge: string; // 场景等级: "A" | "B" | "D"
  title: string; // 场景标题，如 "独立分析 / 策略研究"
  tag: string; // 状态标签: "立即启动" | "继续保持" | "不推荐"
}

interface ScenarioDetail {
  title: string; // 详情标题，如 "场景 1：写作与记录思考"
  desc: string; // 详细描述
}

// ============================================================
// action - 进化路径与行动
// ============================================================
interface Action {
  title: string; // 固定值: "进化路径与行动"
  subtitle: string; // 固定值: "Action Roadmap"
  weekly: WeeklyPlan;
  roadmap: Roadmap;
}

interface WeeklyPlan {
  title: string; // 固定值: "本周行动 (7 Days Plan)"
  items: WeeklyItem[]; // 固定2项
}

interface WeeklyItem {
  index: string; // 序号: "1" 或 "2"
  title: string; // 行动标题（以冒号结尾），如 "案例收集："
  desc: string; // 行动描述
}

interface Roadmap {
  title: string; // 固定值: "90天计划 (90 Days Roadmap)"
  items: RoadmapItem[]; // 固定3项: M1 → M2 → M3
}

interface RoadmapItem {
  title: string; // 阶段标题，格式: "M1: xxx" | "M2: xxx" | "M3: xxx"
  desc: string; // 阶段描述
}

// ============================================================
// growth - 成长建议
// ============================================================
interface Growth {
  title: string; // 固定值: "成长建议"
  subtitle: string; // 固定值: "Growth Suggestions"
  items: GrowthItem[]; // 固定2项
  quote: Quote;
}

interface GrowthItem {
  title: string; // 建议标题，如 "警惕"这不算什么"综合征"
  desc: string; // 建议描述
}

interface Quote {
  lines: string[]; // 金句内容（数组，支持多行）
  signature: string; // 署名，如 "閆婷婷 · 幕后军师"
}

// ============================================================
// footer - 页脚
// ============================================================
interface Footer {
  text: string; // 固定值: "Talent Intelligence Report © 2026"
}

// ============================================================
// cta - 底部行动按钮
// ============================================================
interface CTA {
  text: string; // 固定值: "保存并分享我的天赋卡片"
}
```
